### Kaliningrad

STATE_KRALOVEC_state_name_assign = {
	if = {
        limit = {
            owner = {
                any_primary_culture = {
					cu:polish ?= this
                }
            }
        }
        set_state_name = STATE_KRALOVEC_polish
        set_hub_names = polish
    }
	else_if = {
        limit = {
			owner = {
				country_has_primary_culture = cu:north_german
			}
        }
        set_state_name = STATE_KRALOVEC_german
		set_hub_names = german
    }
	else_if = {
		limit = {
			owner = {
				and = {
					country_definition = {
						any_primary_culture = {
							shares_heritage_trait_with_any_primary_culture = c:RUS
						}
					}
					NOT = {
						any_primary_culture = {
							cu:polish ?= this
						}
					}
				}
			}
		}
		
		set_state_name = STATE_KRALOVEC_russian
		reset_hub_names = yes # Hubs are Russian by default
	}
	else = { # By putting an else at the end that resets the names we ensure that the names are switched back to their defaults if anyone else takes the states
		reset_state_name = yes
		reset_hub_names = yes
	}
}
