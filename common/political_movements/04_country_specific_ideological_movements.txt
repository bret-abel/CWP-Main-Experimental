movement_nuclear_disarmament = {
	category = movement_category_ideological
	
	ideology = ideology_pacifist
	character_ideologies = {
		ideology_pacifist
	}
	
	creation_trigger = {
		has_modifier = atomic_capable
	}
	creation_weight = {
		value = 100
	}	
	on_created = {
		add_movement_enthusiasm_modifier = yes
	}
	
	character_support_trigger = {
		OR = {
			has_ideology = ideology:ideology_pacifist
		}
	}
	character_support_weight = {
		value = 200
		if = {
			limit = {
				has_ideology = ideology:ideology_pacifist
			}
			multiply = {
				value = 5
			}
		}
	}

	pop_support_trigger = {
		OR = {
			is_pop_type = clericals
			is_pop_type = clergy
			is_pop_type = engineers
			is_pop_type = soldiers
			is_pop_type = laborers
			is_pop_type = landlords
			is_pop_type = informal_workers
		}
		NOR = {
			is_pop_type = administrators
			is_pop_type = officers
		}
	}

	pop_support_factors = {
		movement_support_academics
	}
	
	pop_support_weight = {
		value = 1

		if = {
			limit = {
				is_pop_type = clericals
			}
			add = {
				desc = "POP_CLERICALS"
				value = 9
			}
		}
		else_if = {
			limit = {
				is_pop_type = clergy
			}
			add = {
				desc = "POP_CLERGY"
				value = 6
			}
		}
		else_if = {
			limit = {
				is_pop_type = engineers
			}
			add = {
				desc = "POP_ENGINEERS"
				value = 3
			}
		}
		else_if = {
			limit = {
				is_pop_type = soldiers
			}
			add = {
				desc = "POP_SOLDIERS"
				value = 3
			}
		}
		else_if = {
			limit = {
				is_pop_type = landlords
			}
			add = {
				desc = "POP_LABORERS"
				value = 3
			}
		}
		else_if = {
			limit = {
				is_pop_type = informal_workers
			}
			add = {
				desc = "POP_LABORERS"
				value = 3
			}
		}

		multiply = { # Illiterate pops participate at 1/10 the rate of fully literate pops
			desc = "POP_LITERACY"
			value = literacy_rate
			add = 0.10
		}

		multiply = {
			desc = "POP_ACCEPTANCE"
			value = pop_acceptance
			divide = 75
		}
	}

	revolution = {
		possible = {
			political_movement_support >= 0.05
		}
	
		weight = {
			value = 100
		}
	
		state_weight = {
			value = 1
			if = {
				limit = { 
					OR = {
					    any_scope_building = {
							is_building_group = bg_bureaucracy
							level >= 5
					    }		
					}
				}
				multiply = 10
			}
			else_if = {
				limit = { 
					OR = {
					    any_scope_building = {
							is_building_group = bg_nuclear_weapons
							level >= 1
					    }		
					}
				}
				multiply = 10
			}
			else_if = {
				limit = { 
					OR = {
					    any_scope_building = {
							is_building_group = bg_nuclear_research
							level >= 1
					    }		
					}
				}
				multiply = 5
			}
			else = {
				multiply = 0
			}
		}
	}
	
	secession = {
		possible = {
			always = no
		}
		
		weight = {
			value = 0
		}		
		
		state_weight = {
			value = 1
		}		
	}	
	
	additional_radicalism_factors = {
		owner = {
			if = {
				limit = { has_institution = institution_schools }
				
				add = {
					value = 3
					subtract = "institution_investment_level(institution_schools)"
					multiply = 0.05
					desc = "INSTITUTION_FUNDING_LEVEL_schools"
				}
			}
			else = {
				add = {
					value = 0.1
					desc = "NO_INSTITUTION_schools"
				}		
			}				
		}
	}
}
